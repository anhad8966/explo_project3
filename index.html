import numpy as np
import matplotlib.pyplot as plt

def fourbar_position(L1, L2, L3, L4, theta2_rad):
    """
    Solve four-bar position using Freudenstein's equation.
    Returns (theta3, theta4). Raises ValueError at toggle positions.
    """
    K1 = L1 / L4
    K2 = L1 / L2
    K3 = (L2**2 - L3**2 + L4**2 + L1**2) / (2 * L2 * L4)

    A = np.cos(theta2_rad) - K1 - K2 * np.cos(theta2_rad) + K3
    B = -2 * np.sin(theta2_rad)
    C = K1 - (K2 + 1) * np.cos(theta2_rad) + K3

    disc = B**2 - 4 * A * C
    if disc < 0:
        raise ValueError("Toggle position — mechanism cannot assemble.")

    # Assembly Mode: 1 for Open (Upward), -1 for Crossed (Downward)
    mode = 1 
    theta4 = 2 * np.arctan((-B + mode * np.sqrt(disc)) / (2 * A))

    Ax = L1 + L4 * np.cos(theta4) - L2 * np.cos(theta2_rad)
    Ay = L4 * np.sin(theta4) - L2 * np.sin(theta2_rad)
    theta3 = np.arctan2(Ay, Ax)

    return theta3, theta4


def fourbar_velocity(L2, L3, L4, theta2, theta3, theta4, omega2):
    """ Velocity analysis via matrix method. """
    A = np.array([
        [-L3 * np.sin(theta3),  L4 * np.sin(theta4)],
        [ L3 * np.cos(theta3), -L4 * np.cos(theta4)]
    ])
    b = np.array([
         L2 * omega2 * np.sin(theta2),
        -L2 * omega2 * np.cos(theta2)
    ])
    omega3, omega4 = np.linalg.solve(A, b)
    return omega3, omega4


def fourbar_acceleration(L2, L3, L4, theta2, theta3, theta4, omega2, omega3, omega4, alpha2=0.0):
    """ Acceleration analysis via matrix method. """
    A = np.array([
        [-L3 * np.sin(theta3),  L4 * np.sin(theta4)],
        [ L3 * np.cos(theta3), -L4 * np.cos(theta4)]
    ])
    rhs1 = (L2 * alpha2 * np.sin(theta2) + L2 * omega2**2 * np.cos(theta2)
            + L3 * omega3**2 * np.cos(theta3) - L4 * omega4**2 * np.cos(theta4))
    rhs2 = (-L2 * alpha2 * np.cos(theta2) + L2 * omega2**2 * np.sin(theta2)
             + L3 * omega3**2 * np.sin(theta3) - L4 * omega4**2 * np.sin(theta4))
    alpha3, alpha4 = np.linalg.solve(A, np.array([rhs1, rhs2]))
    return alpha3, alpha4
