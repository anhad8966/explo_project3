import sys
import numpy as np

# ==========================================
# Problem 1: Slider-Crank Analysis
# ==========================================
# --- Input Section ---
r = float(input("Enter crank length (r): "))
l = float(input("Enter connecting rod length (l): "))

# Condition to exit if the mechanism cannot be formed/rotate fully
if l <= r:
    sys.exit(f"Error: Connecting rod (l={l}) must be greater than crank radius (r={r}). "
             "Mechanism cannot be formed or will lock.")

omega = float(input("Enter crank angular velocity (rad/s): "))
angles = input("Enter crank angles in degrees (separated by space): ").split()

def slider_crank_kinematics(r, l, omega, theta_deg):
    theta = np.radians(theta_deg)
    n = l / r
    
    cos_theta = np.cos(theta)
    sin_theta = np.sin(theta)
    denom = np.sqrt(n**2 - sin_theta**2)
    
    omega_2 = (omega * cos_theta) / denom
    v_s = -r * omega * (sin_theta + (sin_theta * cos_theta) / denom)
    alpha_2 = (-omega**2 * sin_theta * (n**2 - 1)) / (denom**3)
    a_s = -r * (omega**2 * cos_theta + (omega**2 * (cos_theta**2 - sin_theta**2) / denom) + (omega**2 * sin_theta**2 * cos_theta**2) / (denom**3))
    
    return omega_2, alpha_2, v_s, a_s

print(f"\n{'Angle (°)':<10} {'ω_rod':<15} {'α_rod':<15} {'v_slider':<12} {'a_slider':<12}")
print("-" * 70)

for angle in angles:
    th = float(angle)
    w2, a2, vs, as_val = slider_crank_kinematics(r, l, omega, th)
    print(f"{th:<10} {w2:<15.4f} {a2:<15.4f} {vs:<12.4f} {as_val:<12.4f}")

# ==========================================
# Problem 2: Four-Bar Analysis
# ==========================================
# -----------------------------
# INPUT
# -----------------------------
L1 = float(input("\nEnter L1: "))
L2 = float(input("Enter L2: "))
L3 = float(input("Enter L3: "))
L4 = float(input("Enter L4: "))

omega2 = float(input("Enter ω2 (rad/s): "))

theta2_list = list(map(float, input("Enter θ2 (deg): ").split()))
theta2_list = np.radians(theta2_list)

# -----------------------------
# VALIDATION
# -----------------------------
if min(L1, L2, L3, L4) <= 0:
    raise ValueError("All link lengths must be positive")

print("\n===== RESULTS (FORMULA METHOD) =====\n")

# -----------------------------
# MAIN LOOP
# -----------------------------
for theta2 in theta2_list:

    # ---- POSITION ----
    K1 = L1 / L2
    K2 = L1 / L4
    K3 = (L2**2 - L3**2 + L4**2 + L1**2) / (2 * L2 * L4)

    A = np.cos(theta2) - K1 - K2*np.cos(theta2) + K3
    B = -2*np.sin(theta2)
    C = K1 - (K2 + 1)*np.cos(theta2) + K3

    D = B**2 - 4*A*C

    if D < 0:
        print(f"θ2 = {np.degrees(theta2):.2f}° → No real configuration ")
        continue

    # θ4
    theta4 = 2*np.arctan((-B - np.sqrt(D)) / (2*A))

    # θ3
    num = L4*np.sin(theta4) - L2*np.sin(theta2)
    den = L1 + L4*np.cos(theta4) - L2*np.cos(theta2)

    theta3 = np.arctan2(num, den)

    # ---- VELOCITY ----
    denom_v = np.sin(theta3 - theta4)

    if abs(denom_v) < 1e-6:
        print(f"θ2 = {np.degrees(theta2):.2f}° → Toggle position")
        continue

    omega3 = (L2 * omega2 * np.sin(theta4 - theta2)) / (L3 * denom_v)

    omega4 = (L2 * omega2 * np.sin(theta2 - theta3)) / (L4 * np.sin(theta4 - theta3))

    # ---- ACCELERATION ----
    A1 = (-L2*omega2**2*np.cos(theta2 -theta4) -
          L3*omega3**2*np.cos(theta3 - theta4) +
          L4*omega4**2)

    A2 = (-L2*omega2**2*np.cos(theta2 - theta3) -
          L3*omega3**2 +
          L4*omega4**2*np.cos(theta4 -theta3))

    alpha3 = A1 / (L3 * denom_v)

    alpha4 = A2 / (L4 * np.sin(theta3 - theta4))

    # ---- OUTPUT ----
    print("===================================")
    print(f"θ2 = {np.degrees(theta2):.2f}°")
    print(f"θ3 = {np.degrees(theta3):.2f}°, θ4 = {np.degrees(theta4):.2f}°")

    print(f"ω3 = {omega3:.4f} rad/s, ω4 = {omega4:.4f} rad/s")
    print(f"α3 = {alpha3:.4f} rad/s², α4 = {alpha4:.4f} rad/s²")
